  + identities : List<PARTY_IDENTITY> [1..*]
  + contacts : List<CONTACT> [*]

  + reverse_relationships : List<LOCATABLE_REF> [*]
  + uid : HIER_OBJECT_ID [1]
  + relationships : List<PARTY_RELATIONSHIP> [*]
  + type() : ITEM_STRUCTURE [1]
}

}
