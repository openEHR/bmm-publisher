class CODE_SET {
  + name : String [1]
  + openehr_id : String [1]
  + issuer : String [1]
  + codes : List<CODE> [*]

  + status : TERMINOLOGY_STATUS [0..1]
}

  + name : String [1]
  + concepts : List<TERMINOLOGY_CONCEPT> [*]

}

  + value : String [1]

}

  + id : String [1]
  + rubric : String [1]

  + language : String [1]

  + vocabularies : List<TERMINOLOGY_GROUP> [*]

  active 
  retired 
}
