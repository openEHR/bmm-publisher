@startuml

abstract class EVENT <T> {
  + time : DV_DATE_TIME [1]
  + state : ITEM_STRUCTURE [0..1]
  + data : ITEM_STRUCTURE [1]
  + offset() : DV_DURATION [1]
}

class POINT_EVENT <T> {
}

class INTERVAL_EVENT <T> {
  + width : DV_DURATION [1]
  + sample_count : Integer [0..1]
  + math_function : DV_CODED_TEXT [1]
  + interval_start_time() : DV_DATE_TIME [1]
}

class HISTORY <T> {
  + origin : DV_DATE_TIME [1]
  + period : DV_DURATION [0..1]
  + duration : DV_DURATION [0..1]
  + summary : ITEM_STRUCTURE [0..1]
  + events : List<EVENT<T>> [*]
  + is_periodic() : Boolean [1]
}

LOCATABLE <|-- EVENT
EVENT <|-- POINT_EVENT
EVENT <|-- INTERVAL_EVENT
DATA_STRUCTURE <|-- HISTORY

@enduml
