class BMM_CONDITIONAL_ACTION {
  + condition : EL_BOOLEAN_EXPRESSION [1]
  + statement : BMM_STATEMENT_ITEM [1]
}

class BMM_ACTION_TABLE {
  + items : List<BMM_CONDITIONAL_ACTION> [1..*]
}

class BMM_ASSIGNMENT {
  + target : EL_INSTANTIABLE_REF [1]
  + source : EL_EXPRESSION [1]
}

class BMM_ASSERTION {
  + expression : EL_BOOLEAN_EXPRESSION [1]
  + tag : String [0..1]
}

class BMM_PROCEDURE_CALL {
  + agent : EL_PROCEDURE_AGENT [1]
}

abstract class BMM_STATEMENT_ITEM {
}

abstract class BMM_STATEMENT {
}

class BMM_STATEMENT_BLOCK {
  + items : List<BMM_STATEMENT_ITEM> [*]
}

abstract class BMM_SIMPLE_STATEMENT {
}

BMM_STATEMENT <|-- BMM_ACTION_TABLE

BMM_SIMPLE_STATEMENT <|-- BMM_ASSERTION
EL_AGENT_CALL <|-- BMM_PROCEDURE_CALL
BMM_SIMPLE_STATEMENT <|-- BMM_PROCEDURE_CALL
BMM_ROUTINE_BODY <|-- BMM_STATEMENT_ITEM
BMM_STATEMENT_ITEM <|-- BMM_STATEMENT
BMM_STATEMENT_ITEM <|-- BMM_STATEMENT_BLOCK
BMM_STATEMENT <|-- BMM_SIMPLE_STATEMENT
