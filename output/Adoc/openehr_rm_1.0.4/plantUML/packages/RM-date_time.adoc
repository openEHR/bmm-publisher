  + is_strictly_comparable_to( other : DV_ORDERED [1] ) : Boolean [1]

}

class DV_DATE {
  + value : String [1]
  + magnitude() : Integer [1]
  + is_strictly_comparable_to( other : DV_ORDERED [1] ) : Boolean [1]
  + infix <( other_quantity : DV_ORDERED [1] ) : Boolean [1]
  + add( a_diff : DV_AMOUNT [1] ) : DV_ABSOLUTE_QUANTITY [1]
  + subtract( a_diif : DV_AMOUNT [1] ) : DV_ABSOLUTE_QUANTITY [1]
  + diff( other : DV_DATE [1] ) : DV_DURATION [1]
}

class DV_TIME {
  + value : String [1]
  + magnitude() : Real [1]
  + is_strictly_comparable_to( other : DV_ORDERED [1] ) : Boolean [1]
  + add( a_diff : DV_AMOUNT [1] ) : DV_ABSOLUTE_QUANTITY [1]
  + subtract( a_diif : DV_AMOUNT [1] ) : DV_ABSOLUTE_QUANTITY [1]
  + operator=( source : DV_TIME [1] ) : DV_TIME [1]
  + diff( other : DV_TIME [1] ) : DV_DURATION [1]
}

class DV_DATE_TIME {
  + value : String [1]
  + magnitude() : Double [1]
  + is_strictly_comparable_to( other : DV_ORDERED [1] ) : Boolean [1]
